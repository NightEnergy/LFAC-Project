int globalVar;

class Calculator {
    int factor;
    
    int multiply(int val) {
        return val * factor;
    }

    void setFactor(int f) {
        factor <- f;
    }
};

int factorial(int n) {
    if (n <= 1) return 1;
    return n * factorial(n - 1);
}

begin_progr
    globalVar <- 10;
    
    Calculator calc;
    calc.setFactor(2);
    
    int result;
